\begindesc
\cts splitmaxdepth {\param{dimen}}
\explain
This parameter specifies the
maximum allowable depth of a box resulting from a |\vsplit|.
|\splitmaxdepth| plays the
same role that ^|\maxdepth| (\xref \maxdepth) plays for a page.
\enddesc

\enddescriptions
