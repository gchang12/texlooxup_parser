\begindesc
\cts splittopskip {\param{glue}}
\explain
This parameter specifies the glue that \TeX\ inserts at the top of
a box resulting from a |\vsplit|.
|\splittopskip| plays the
same role that ^|\topskip| (\xref \topskip) plays for a page.
\enddesc

\enddescriptions
