\input macros

\begindescriptions

\begindesc
^^{boxes//empty}
\cts null {}
\explain
This command produces an empty \minref{hbox}.
\example
\setbox0 = \null
The null box \null has width \the\wd0, height \the\ht0,
and depth \the\dp0.
|
\produces
\setbox0 = \null
The null box \null has width \the\wd0, height \the\ht0,
and depth \the\dp0.
\endexample
\enddesc

\enddescriptions
