\input macros
\begindescriptions
\begindesc
\easy\cts root {\<argument$_1$> {\bt \\of} \<argument$_2$>}
\explain
This command produces the notation for a root of \<argument$_2$>, where the
root is given by \<argument$_1$>.
\example
$\root \alpha \of {r \cos \theta}$
|
\produces
$\root \alpha \of {r \cos \theta}$
\endexample
\enddesc
\enddescriptions
