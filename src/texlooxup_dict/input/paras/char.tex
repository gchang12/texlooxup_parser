\input macros
\begindescriptions
\begindesc
\bix^^{characters}
\cts char {\<charcode>}
\explain
This command produces the character located at position \<charcode>
of the current font.
\example
{\char65} {\char `A} {\char `\A}
|
\produces
{\char65} {\char `A} {\char `\A}
\endexample
\enddesc
\enddescriptions
\end
