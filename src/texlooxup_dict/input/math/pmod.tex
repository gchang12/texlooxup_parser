\input macros
\begindescriptions
\begindesc
\cts pmod {}
\explain
This command provides a notation for indicating a ^{modulus} in parentheses
at the end of a formula.
\example
$$x \equiv y+1 \pmod 2$$
|
\dproduces
$$x \equiv y+1 \pmod 2$$

\eix^^{functions, names of}
\endexample
\enddesc
\enddescriptions
\end
