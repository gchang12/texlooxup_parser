\input macros
\begindescriptions
\begindesc
\cts underbar {\<argument>}
\explain
This command puts \<argument> into an \minref{hbox}
and underlines it without regard to anything that protrudes below the
\minref{baseline} of the box.
\example
\underbar{Why not learn \TeX?}
|
\produces
\underbar{Why not learn \TeX?}
\endexample
\enddesc
\enddescriptions
